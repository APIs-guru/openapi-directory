## SDK Example Usage

```typescript
import { SDK, WithSecurity} from "openapi";
import { DipcrRequest, DipcrResponse } from "openapi/src/sdk/models/operations";
import { AxiosError } from "axios";


const sdk = new SDK();
    
const req: DipcrRequest = {
  security: {
    apiKey: {
      apiKey: "YOUR_API_KEY_HERE",
    }
    clientId: {
      apiKey: "YOUR_API_KEY_HERE",
    },
  },
  request: {
    certificateParameters: {
      fullName: "iste",
      rollNo: "nulla",
    },
    consentArtifact: "autem",
    format: "pdf",
    txnId: "recusandae",
  },
};

sdk.sdk.dipcr(req).then((res: DipcrResponse | AxiosError) => {
   // handle response
});
```
