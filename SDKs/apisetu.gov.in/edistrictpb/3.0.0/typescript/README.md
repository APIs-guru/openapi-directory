## SDK Example Usage

```typescript
import { SDK, WithSecurity} from "openapi";
import { BtcerRequest, BtcerResponse } from "openapi/src/sdk/models/operations";
import { AxiosError } from "axios";


const sdk = new SDK();
    
const req: BtcerRequest = {
  security: {
    apiKey: {
      apiKey: "YOUR_API_KEY_HERE",
    }
    clientId: {
      apiKey: "YOUR_API_KEY_HERE",
    },
  },
  request: {
    certificateParameters: {
      dob: "ab",
      registrationId: "earum",
    },
    consentArtifact: "dolores",
    format: "pdf",
    txnId: "exercitationem",
  },
};

sdk.sdk.btcer(req).then((res: BtcerResponse | AxiosError) => {
   // handle response
});
```
