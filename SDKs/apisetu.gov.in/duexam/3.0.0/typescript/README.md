    certificateParameters: {

      rroll: "sint",
      year: "fuga",
    },
    consentArtifact: "pariatur",
    format: "pdf",
    txnId: "ducimus",
  },

sdk.sdk.dgcer(req).then((res: DgcerResponse | AxiosError) => {
   // handle response
});
