## SDK Example Usage

```typescript
import { SDK, WithSecurity} from "openapi";
import { SocerRequest, SocerResponse } from "openapi/src/sdk/models/operations";
import { AxiosError } from "axios";


const sdk = new SDK();
    
const req: SocerRequest = {
  security: {
    apiKey: {
      apiKey: "YOUR_API_KEY_HERE",
    }
    clientId: {
      apiKey: "YOUR_API_KEY_HERE",
    },
  },
  request: {
    certificateParameters: {
      applNo: "et",
      dob: "et",
      fullName: "quia",
      session: "et",
    },
    consentArtifact: "maiores",
    format: "pdf",
    txnId: "velit",
  },
};

sdk.sdk.socer(req).then((res: SocerResponse | AxiosError) => {
   // handle response
});
```
