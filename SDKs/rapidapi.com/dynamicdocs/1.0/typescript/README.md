## SDK Example Usage

```typescript
import { SDK, WithSecurity} from "openapi";
import { CompileRequest, CompileResponse } from "openapi/src/sdk/models/operations";
import { AxiosError } from "axios";

const sdk = new SDK(WithSecurity(
  security: {
    advSecurityToken: {
      apiKey: "YOUR_API_KEY_HERE",
    },
  }
));
    
const req: CompileRequest = {
  pathParams: {
    templateToken: "qui",
  },
  queryParams: {
    docFileName: "et",
    docUrlExpiresIn: 25486664550251904,
    latexCompiler: "lualatex",
    latexRuns: 8784083526164144924,
    mainFileName: "expedita",
  },
  headers: {
    contentType: "ipsa",
  },
  request: {
    "neque": "repellendus",
    "et": "atque",
  },
};

sdk.pdfGeneration.compile(req).then((res: CompileResponse | AxiosError) => {
   // handle response
});
```
