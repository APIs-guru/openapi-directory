## SDK Example Usage

```typescript
import { SDK, WithSecurity} from "openapi";
import { DeleteProxyRequest, DeleteProxyResponse } from "openapi/src/sdk/models/operations";
import { AxiosError } from "axios";

const sdk = new SDK(WithSecurity(
  security: {
    apiKey: {
      apiKey: "YOUR_API_KEY_HERE",
    }
    applicationId: {
      apiKey: "YOUR_API_KEY_HERE",
    },
  }
));
    
const req: DeleteProxyRequest = {
  security: {
    apiKey: {
      apiKey: "YOUR_API_KEY_HERE",
    },
  },
  headers: {
    xApideckAppId: "ut",
    xApideckConsumerId: "sint",
    xApideckDownstreamAuthorization: "optio",
    xApideckDownstreamUrl: "aut",
    xApideckServiceId: "eos",
  },
};

sdk.execute.deleteProxy(req).then((res: DeleteProxyResponse | AxiosError) => {
   // handle response
});
```
