## SDK Example Usage

```typescript
import { SDK, WithSecurity} from "openapi";
import { WebhooksAddRequest, WebhooksAddResponse } from "openapi/src/sdk/models/operations";
import { AxiosError } from "axios";

const sdk = new SDK(WithSecurity(
  security: {
    apiKey: {
      apiKey: "YOUR_API_KEY_HERE",
    },
  }
));
    
const req: WebhooksAddRequest = {
  security: {
    apiKey: {
      apiKey: "YOUR_API_KEY_HERE",
    },
  },
  headers: {
    xApideckAppId: "quidem",
  },
  request: {
    deliveryUrl: "incidunt",
    description: "ratione",
    events: [
      "crm.lead.updated",
    ],
    status: "enabled",
    unifiedApi: "transactional-email",
  },
};

sdk.webhooks.webhooksAdd(req).then((res: WebhooksAddResponse | AxiosError) => {
   // handle response
});
```
